23  public class ScrollBarControlSteps {
24      private HashMap<String, Win32Locator> scrollBarMap = new HashMap<String, Win32Locator>();
25  
26      public ScrollBarControlSteps() {
27          scrollBarMap.put("Horizontal", new Win32Locator("ScrollBar",
28                  "(.*)", 1));
29          scrollBarMap.put("Vertical", new Win32Locator("ScrollBar",
30                  "(.*)", 0));
31      }

33      private long getControl(String fieldName) throws Exception {
34          Win32Utils utils = new Win32Utils();
35          Win32Locator locator;
36          Win32Locator pageLocator;
37  
38          if (!scrollBarMap.containsKey(fieldName)) {
39              locator = new Win32Locator("ScrollBar", fieldName, 0);
40          } else {
41              locator = scrollBarMap.get(fieldName);
42          }
43          long htab = CommonSteps.getMainTabControl();
44          TabControl tab = new TabControl();
45  
46          pageLocator = new Win32Locator("#32770", tab.GetSelectedItem(htab), 0);
47  
48          long hpage = utils.searchWindow(pageLocator);
49  
50          locator.setParent(hpage);
51          long hctrl = utils.searchWindow(locator);
52          return hctrl;
53      }
54      
55      @When("^(?:I |)set the \"(\\d+)\" position for the \"([^\"]*)\" scroll bar$")
56      public void setPositionByIndex(int position, String fieldName) throws Throwable {
57          int hctrl = (int)getControl(fieldName);

72      @When("^(?:I |)add (\\d+) step (forward|backward) for the \"([^\"]*)\" scroll bar$")
73      public void addStep(int step, String direction, String fieldName) throws Throwable {
74          boolean forward = direction.equals("forward");
75          int hctrl = (int) getControl(fieldName);
76          ScrollBar scroll = new ScrollBar();        
77  
78          for(int i=0;i<step;i++){
79              if(forward){

85          }
