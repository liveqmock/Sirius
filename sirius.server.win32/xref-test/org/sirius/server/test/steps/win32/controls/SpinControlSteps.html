29      private long getControl(String fieldName) throws Exception {
30          Win32Utils utils = new Win32Utils();
31          Win32Locator locator;
32          Win32Locator pageLocator;
33  
34          if (!spinMap.containsKey(fieldName)) {
35              locator = new Win32Locator("Edit", fieldName, 0);
36          } else {
37              locator = spinMap.get(fieldName);
38          }
39          long htab = CommonSteps.getMainTabControl();
40          TabControl tab = new TabControl();
41  
42          pageLocator = new Win32Locator("#32770", tab.GetSelectedItem(htab), 0);
43  
44          long hpage = utils.searchWindow(pageLocator);
45  
46          locator.setParent(hpage);
47          long hctrl = utils.searchWindow(locator);
48          return hctrl;
49      }
50  
51      @When("^(?:I )increment \"([^\"]*)\" field value$")
52      public void incrementSpinValue(String fieldName) throws Throwable {
53          int hctl = (int) getControl(fieldName);
